📋 Schema SQL Supabase

Exécutez ce SQL dans votre projet Supabase pour créer les tables nécessaires :

CREATE TABLE IF NOT EXISTS funnel_config (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  funnel_id text UNIQUE NOT NULL,
  config jsonb NOT NULL DEFAULT '{}',
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS funnel_events (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  funnel_id text NOT NULL,
  visitor_id text NOT NULL,
  event_type text NOT NULL,
  event_data jsonb DEFAULT '{}',
  page_url text,
  user_agent text,
  referrer text,
  created_at timestamptz DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_events_funnel_created
  ON funnel_events(funnel_id, created_at);
CREATE INDEX IF NOT EXISTS idx_events_type
  ON funnel_events(event_type);
